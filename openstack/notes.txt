# OpenStack

Initial documentation for doing kolla-ansible in the minilab configuration


## Initial Setup
source .env/bin/activate
kolla-ansible bootstrap-servers -i lab
kolla-ansible prechecks -i lab
kolla-ansible deploy -i lab
kolla-ansible post-deploy -i lab

## Tear Down
kolla-ansible destroy -i lab --yes-i-really-really-mean-it

## New Controller (untested)
kolla-ansible bootstrap-servers -i lab --limit <NEW_CONTROLLER>
kolla-ansible pull -i lab --limit <NEW_CONTROLLER>
kolla-ansible deploy -i lab --limit control

## New Compute 
- Add it to the [compute] section
kolla-ansible bootstrap-servers -i lab --limit <NEW_COMPUTE>
kolla-ansible pull -i lab --limit <NEW_COMPUTE>
kolla-ansible deploy -i lab --limit <NEW_COMPUTE>

## Live migration to new Compute Node (untested)
openstack server migrate <TARGET_COMPUTE> --live-migration --host <NEW_COMPUTE> --os-compute-api-version 2.30

## Remove Compute
source /etc/kolla/admin-openrc.sh
export OS_CLIENT_CONFIG_FILE=/etc/kolla/clouds.yaml
openstack compute service list
openstack compute service set <REMOVE_COMPUTE> nova-compute --disable
## If there are resources being used
openstack server list --all-projects --host <REMOVE_COMPUTE> -f value -c ID | while read server; do
  openstack server migrate --live-migration <EXISTING_COMPUTE>
done
kolla-ansible stop -i lab --yes-i-really-really-mean-it  --limit <REMOVE_COMPUTE>
openstack network agent list --host <REMOVE_COMPUTE> -f value -c ID | while read id; do
  openstack network agent delete $id
done
openstack compute service list --os-compute-api-version 2.53 --host <REMOVE_COMPUTE> -f value -c ID | while read id; do
  openstack compute service delete --os-compute-api-version 2.53 $id
done
